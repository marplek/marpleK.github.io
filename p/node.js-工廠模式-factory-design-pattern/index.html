    log(message) {

        console.log(`\x1b[32m${message}\x1b[0m`);

function createColorConsole(color) {

		    return new RedConsole();
	    case 'blue':
		    return new BlueConsole();
	    case 'green':
		    return new GreenConsole();
		default:

	}
}
